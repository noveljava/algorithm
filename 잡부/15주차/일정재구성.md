import java.util.ArrayDeque
import java.util.Deque
import java.util.LinkedList
import java.util.PriorityQueue

class ReconstructItinerary {

    fun findItinerary(tickets: List<List<String>>): List<String> {
        val fromToMap: MutableMap<String, PriorityQueue<String>> = mutableMapOf()
        for (ticket in tickets) {
            val from = ticket[0]
            val to = ticket[1]
            if (!fromToMap.containsKey(from)) {
                fromToMap[from] = PriorityQueue()
            }
            fromToMap[from]!!.add(to)
        }

        val results: MutableList<String> = LinkedList()
        val stack: Deque<String> = ArrayDeque()

        stack.push("ICN")
        while (stack.isNotEmpty()) {
            val from = stack.peek()
            if (fromToMap.containsKey(from) && fromToMap[from]!!.isNotEmpty()) {
                stack.push(fromToMap[from]!!.poll())
            } else {
                results.add(0, stack.pop())
            }
        }

        return results
    }
}
